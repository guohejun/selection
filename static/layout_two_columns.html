float:left+margin-left

.parent1 .left { float: left; width: 300px; height: 100%; /*没有达到100%,会导致right侵入到left下方*/ } .parent1 .right { height: 100%; margin-left: 300px; }

float:left+overflow

.parent2 .left { width: 300px; float: left; height: 100%; } .parent2 .right { overflow: hidden; height: 100%; }

float:left + float:right + calc

.parent3 .left { width: 300px; float: left; height: 100%; } .parent3 .right { width: calc(100% - 300px); float: right; height: 100%; }

flex

.parent4 { display: flex; } .parent4 .left { width: 300px; height: 100%; } .parent4 .right { flex: 1; height: 100%; }
